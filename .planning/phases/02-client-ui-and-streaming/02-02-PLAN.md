  text: string;
  timestamp: string;
  isStreaming?: boolean;
}

  id: string;
  contextId: string;
  status: TaskState;
  messages: ChatMessage[];
  rawExchanges: RawExchange[];

  id: string;
  request?: unknown;
  response?: unknown;
  timestamp: string;
}

  status: ConnectionStatus;
  sseStatus: SSEStatus;
  agentCard: AgentCardInfo | null;
  remoteUrl: string;
  tasks: Map<string, TaskData>;

}
